# SSH vao VPS
ssh -o StrictHostKeyChecking=no root@160.191.88.175

# Kiem tra Nginx site dang enable
ls -la /etc/nginx/sites-enabled

# Xem cau hinh default frontend
cat /etc/nginx/sites-available/frontend-default

# Kiem tra DNS domain
getent hosts demostrapi.webcanhcam.vn
nslookup -type=A demostrapi.webcanhcam.vn

# Kiem tra container dang chay cong nao
docker ps --format 'table {{.Names}}\t{{.Image}}\t{{.Ports}}'

# Tao file cau hinh Nginx cho domain moi (HTTP + HTTPS)
cat > /etc/nginx/sites-available/demostrapi.webcanhcam.vn <<'NGINX_EOF'
server {
    listen 80;
    listen [::]:80;
    server_name demostrapi.webcanhcam.vn;

    location / {
        proxy_pass http://127.0.0.1:3100;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}

server {
    listen 443 ssl;
    listen [::]:443 ssl;
    server_name demostrapi.webcanhcam.vn;

    ssl_certificate /etc/letsencrypt/live/strapi.webcanhcam.vn/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/strapi.webcanhcam.vn/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://127.0.0.1:3100;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
NGINX_EOF

# Enable site
ln -sfn /etc/nginx/sites-available/demostrapi.webcanhcam.vn /etc/nginx/sites-enabled/demostrapi.webcanhcam.vn

# Test config + reload nginx
nginx -t
systemctl reload nginx

# Verify ket qua
curl -I -H 'Host: demostrapi.webcanhcam.vn' http://127.0.0.1
curl -I http://demostrapi.webcanhcam.vn
curl -I https://demostrapi.webcanhcam.vn
